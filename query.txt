--CRIAÇÃO DE TABELAS
CREATE TABLE  cliente_locadora (
	id_cliente int identity(1,1) primary key,
    nome varchar(40),
	cpf varchar(40),
	email varchar(40),
	telefone varchar(40)
);

CREATE TABLE  carro_locadora (
	id_carro int identity(1,1) primary key,
    placa varchar(40),
	marca varchar(40),
	cor varchar(40),
	preco varchar(40)
);

CREATE TABLE  moto_locadora (
	id_moto int identity(1,1) primary key,
    marca varchar(40),
	preco varchar(40),
	cor varchar(40),
	ano varchar(40)
);
